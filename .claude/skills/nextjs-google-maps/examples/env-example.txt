# Google Maps API Configuration
# このファイルを .env.local としてコピーして使用してください
# cp env-example.txt .env.local

# ==============================================
# 必須設定
# ==============================================

# Google Maps API キー
# 取得方法: https://console.cloud.google.com/google/maps-apis/credentials
# 注意: NEXT_PUBLIC_ プレフィックスが必要（クライアントサイドで使用するため）
NEXT_PUBLIC_GOOGLE_MAPS_API_KEY=your_api_key_here

# ==============================================
# オプション設定（サーバーサイド API 呼び出し用）
# ==============================================

# サーバーサイド専用 API キー（より厳格な制限を設定可能）
# この変数はクライアントサイドには公開されません
# GOOGLE_MAPS_SERVER_KEY=your_server_api_key_here

# ==============================================
# 開発用設定
# ==============================================

# 開発モードでの詳細ログ出力
# NEXT_PUBLIC_GOOGLE_MAPS_DEBUG=true

# ==============================================
# API キーの設定手順
# ==============================================
#
# 1. Google Cloud Console (https://console.cloud.google.com/) にアクセス
#
# 2. プロジェクトを作成または選択
#
# 3. 以下の API を有効化:
#    - Maps JavaScript API（必須）
#    - Places API（場所検索を使用する場合）
#    - Directions API（ルート計算を使用する場合）
#    - Geocoding API（住所⇔座標変換を使用する場合）
#
# 4. 認証情報 > 認証情報を作成 > API キー
#
# 5. API キーの制限を設定:
#    a. アプリケーションの制限:
#       - 種類: HTTP リファラー（ウェブサイト）
#       - 許可するリファラー:
#         - 開発用: localhost:*
#         - 本番用: https://your-domain.com/*
#
#    b. API の制限:
#       - 使用する API のみを選択
#
# ==============================================
# セキュリティのベストプラクティス
# ==============================================
#
# 1. 本番環境では必ずリファラー制限を設定
# 2. 使用しない API は有効化しない
# 3. 使用量を定期的に監視
# 4. 異常な使用量のアラートを設定
# 5. 開発用と本番用で別の API キーを使用
#
# ==============================================
